/*
 * clears the libnet pblocks from the libnet context
 */

static VALUE rb_libnet_clear_packet(VALUE self)
{
   libnet_t *lctx = NULL;

   Data_Get_Struct(
         rb_iv_get(self, "@handle"),
         libnet_t,
         lctx);

   DEBUG_PRINT("libnet_clear_packet");

   if(lctx)
      libnet_clear_packet(lctx);
   
   return Qnil;
}
