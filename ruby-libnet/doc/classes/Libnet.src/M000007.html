/*
 * returns the last error set inside libnet
 */

static VALUE rb_libnet_geterror(VALUE self)
{
   libnet_t *lctx = NULL;
   char *err;

   Data_Get_Struct(
         rb_iv_get(self, "@handle"),
         libnet_t,
         lctx);

   DEBUG_PRINT("libnet_geterror()");

   err = libnet_geterror(lctx);

   return (err == NULL ? Qnil : rb_str_new2(err));
}
